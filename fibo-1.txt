def fib_recursive(n):
    if n <= 1:
        return n
    else:
        return fib_recursive(n - 1) + fib_recursive(n - 2)

# Driver code
n = int(input("Enter number of terms: "))
print("Fibonacci sequence (Recursive):")
for i in range(n):
    print(fib_recursive(i), end=" ")



def fib_iterative(n):
    a, b = 0, 1
    print("Fibonacci sequence (Non-Recursive):")
    for _ in range(n):
        print(a, end=" ")
        a, b = b, a + b

# Driver code
n = int(input("\nEnter number of terms: "))
fib_iterative(n)
