import heapq

# Class for Huffman Tree Node
class Node:
    def __init__(self, char, freq):
        self.char = char
        self.freq = freq
        self.left = None
        self.right = None

    # Comparison function for priority queue
    def __lt__(self, other):
        return self.freq < other.freq


# Function to print Huffman Codes
def print_codes(node, code=""):
    if node is None:
        return

    # If this is a leaf node, print the code
    if node.char is not None:
        print(f"{node.char} -> {code}")

    print_codes(node.left, code + "0")
    print_codes(node.right, code + "1")


# Function to build Huffman Tree
def huffman_encoding(char_freq):
    heap = [Node(char, freq) for char, freq in char_freq.items()]
    heapq.heapify(heap)

    while len(heap) > 1:
        left = heapq.heappop(heap)
        right = heapq.heappop(heap)

        merged = Node(None, left.freq + right.freq)
        merged.left = left
        merged.right = right

        heapq.heappush(heap, merged)

    return heap[0]  # Root node


# Example usage
if __name__ == "__main__":
    # Example character frequencies
    char_freq = {
        'A': 5,
        'B': 9,
        'C': 12,
        'D': 13,
        'E': 16,
        'F': 45
    }

    root = huffman_encoding(char_freq)
    print("Huffman Codes are:")
    print_codes(root)
